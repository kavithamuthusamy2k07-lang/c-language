#include<stdio.h>
int main()
{
    int a,b,c;
    printf("enter the angles;\n");
    scanf("%d%d%d",&a,&b,&c);
    if((a==b)&&(b==c))
    {
        printf("equilateral triangle");
    }
    else if(a==b&&b!=c)
    {
      printf("isoceleus triangle");
    }
    else if(a!=b&&b!=c&&c!=a)
    {
        printf("scalene triangle");
    }
    return 0;
}
